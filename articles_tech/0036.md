### 计时

### 判断是否是文件夹 dir

```py
import os

def checkDir(prefix):
  list = os.listdir(prefix) # 得到的是当层文件名字符串数组
  for i in list:
    x = prefix + i # 这个地方/需要留意
    if os.path.isdir(x):
      print('[x]', x)
    else:
      print('[ ]', x)

checkDir('./')
```


### 获取本机 ip

```py
import socket

print(socket)

hostname = socket.gethostname()

print(hostname)

ip = socket.gethostbyname(hostname)

print(ip)

s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
s.connect(('8.8.8.8', 80))
r = s.getsockname()
print(r)
print(r[0])
```
